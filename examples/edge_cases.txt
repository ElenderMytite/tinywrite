# Test edge cases and special scenarios

# Negative numbers
neg1 = (-42);
neg2 = (-999);
neg1;
neg2;

# Zero
z = 0;
(+ z 0);
(* z 100);

# Large numbers
big = 999999;
(+ big 1);

# Single character variable names
a = 1;
b = 2;
c = 3;
(+ a b c);

# Underscore in variable names
var_1 = 100;
var_2 = 200;
(+ var_1 var_2);

# Operations with result
(+ 5 3);
(- 20 8);
(* 4 4);
(/ 36 6);

# Comparison chains
(>= 10 5);
(<= 5 10);
(== 42 42);
