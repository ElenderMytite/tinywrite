# Test logical operations
# AND operation
:print (& $true $true);
:print (:char 68);
(& $true $false);

# OR operation
(| $true $false);
(| $false $false);

# XOR operation
(^ $true $false);
(^ $true $true);

# NOT operation
(! $false);
(! $true);

# NAND operation
(!& $true $true);
(!& $true $false);

# NOR operation
(!| $true $false);
(!| $false $false);
